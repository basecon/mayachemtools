  44 def GetMolName(Mol, MolNum = None):
  45     """Get molecule name.
  46     
  47     Arguments:
  48         Mol (object): RDKit molecule object.
  49         MolNum (int or None): Molecule number in input file.
  50 
  51     Returns:
  52         str : Molname corresponding to _Name property of a molecule, generated
  53             from specieid MolNum using the format "Mol%d" % MolNum, or an
  54             empty string.
  55 
  56     """
  57     
  58     MolName = ''
  59     if Mol.HasProp("_Name"):
  60         MolName = Mol.GetProp("_Name")
  61 
  62     if not len(MolName):
  63         if MolNum is not None:
  64             MolName = "Mol%d" % MolNum
  65     
  66     return MolName

 117 def _ModifySVGForInlineEmbedding(SVGText, Base64Encoded):
 118     """Modify SVG for inline embedding into a HTML page using "img" tag
 119     along with performing base64 encoding.
 120     """
 121     
 122     # Take out all tags till the start of '<svg' tag...
 123     Pattern = re.compile("^.*<svg", re.I | re.S)
 124     SVGText = Pattern.sub("<svg", SVGText)
 125     
 126     # Add an extra space before the "width=..." tag. Otherwise, inline embedding may
 127     # cause the following XML error on some browsers due to start of the "width=..."
 128     # at the begining of the line in <svg ...> tag:
 129     #
 130     #  XML5607: Whitespace expected.
 131     #
 132     SVGText = re.sub("width='", " width='", SVGText, flags = re.I)
 133     
 134     # Take out trailing new line...
 135     SVGText = SVGText.strip()
 136 
 137     # Perform base64 encoding by turning text into byte stream using string
 138     # encode and transform byte stream returned by b64encode into a string
 139     # by string decode...
 140     #
 141     if Base64Encoded:
 142         SVGText = base64.b64encode(SVGText.encode()).decode()
 143 
 144     return SVGText

 175 def GetSVGForMolecules(Mols, MolsPerRow, MolWidth, MolHeight, Legends = None, AtomListsToHighlight = None, BondListsToHighlight = None, BoldText = True):
 176     """Get SVG image text for molecules suitable for viewing in a browser.
 177     
 178     Arguments:
 179         Mols (list): List of RDKit molecule objects.
 180         MolsPerRow (int): Number of molecules per row.
 181         Width (int): Width of a molecule image in pixels.
 182         Height (int): Height of a molecule image in pixels.
 183         Legends (list): List containing strings to display under images.
 184         AtomListsToHighlight (list): List of lists containing atoms to highlight
 185             for molecules.
 186         BondListsToHighlight (list): List of lists containing bonds to highlight
 187             for molecules
 188         BoldText (bool): Flag to make text bold in the image of molecules. 
 189 
 190     Returns:
 191         str : SVG image text for writing to a SVG file for viewing in a browser.
 192 
 193     """
 194     
 195     SVGText = Draw.MolsToGridImage(Mols, molsPerRow = MolsPerRow, subImgSize = (MolWidth,MolHeight), legends = Legends, highlightAtomLists = AtomListsToHighlight, highlightBondLists = BondListsToHighlight, useSVG = True)
 196     
 197     return _ModifySVGForBrowserViewing(SVGText, BoldText)

 199 def _ModifySVGForBrowserViewing(SVGText, BoldText = True):
 200     """Modify SVG for loading into a browser."""
 201     
 202     # It appears that the string 'xmlns:svg' needs to be replaced with 'xmlns' in the
 203     # SVG image string generated by RDKit. Otherwise, the image doesn't load in 
 204     # web browsers.
 205     #
 206     if re.search("xmlns:svg", SVGText, re.I):
 207         SVGText = re.sub("xmlns:svg", "xmlns", SVGText, flags = re.I)
 208     
 209     # Make text bold...
 210     if BoldText:
 211         SVGText = re.sub("font-weight:normal;", "font-weight:bold;", SVGText, flags = re.I)
 212     
 213     return SVGText

 249 def AreAtomIndicesSequentiallyConnected(Mol, AtomIndices):
 250     """Check for the presence bonds between sequential pairs of atoms in a
 251     molecule.
 252     
 253     Arguments:
 254         Mol (object): RDKit molecule object.
 255         AtomIndices (list): List of atom indices.
 256 
 257     Returns:
 258         bool : True - Sequentially connected; Otherwise, false. 
 259 
 260     """
 261 
 262     for Index in range(0, (len(AtomIndices) -1)):
 263         if Mol.GetBondBetweenAtoms(AtomIndices[Index], AtomIndices[Index + 1]).GetIdx() is None:
 264             return False
 265     
 266     return True

 281 def MolToBase64EncodedMolString(Mol, PropertyPickleFlags = Chem.PropertyPickleOptions.AllProps):
 282     """Encode RDkit molecule object into a base64 encoded string. The properties
 283     can be optionally excluded.
 284     
 285     The molecule is pickled using RDKit Mol.ToBinary() function before
 286     their encoding.
 287    
 288     Arguments:
 289         Mol (object): RDKit molecule object.
 290         PropertyPickleFlags: RDKit property pickle options.
 291 
 292     Returns:
 293         str : Base64 encode molecule string or None.
 294 
 295     Notes:
 296         The following property pickle flags are currently available in RDKit:
 297             
 298             Chem.PropertyPickleOptions.NoProps
 299             Chem.PropertyPickleOptions.MolProps
 300             Chem.PropertyPickleOptions.AtomProps
 301             Chem.PropertyPickleOptions.BondProps
 302             Chem.PropertyPickleOptions.PrivateProps
 303             Chem.PropertyPickleOptions.AllProps
 304 
 305     """
 306 
 307     return None if Mol is None else base64.b64encode(Mol.ToBinary(PropertyPickleFlags)).decode()

 333 
 334     Returns:
 335         object : Base64 endcoded molecules iterator. The iterator returns a
 336             list containing a molecule index and an encoded molecule string

 339     Notes:

 345             Chem.PropertyPickleOptions.BondProps
 346             Chem.PropertyPickleOptions.PrivateProps

 360 def MolFromSubstructureMatch(Mol, PatternMol, AtomIndices, FilterByAtomMapNums = False):

 365     to appropriate atom indices during the generation of the molecule. For

 367     
 368     Arguments:
 369         Mol (object): RDKit molecule object.
 370         PatternMol (object): RDKit molecule object for a SMILES/SMARTS pattern.
 371         AtomIndices (list): Atom indices.
 372         FilterByAtomMapNums (bool): Filter matches by atom map numbers.
 373 
 374     Returns:
 375         object : RDKit molecule object or None.
 376 
 377     """
 378 
 379     AtomMapIndices = _GetAtomMapIndices(PatternMol) if FilterByAtomMapNums else None

 383 def MolsFromSubstructureMatches(Mol, PatternMol, AtomIndicesList, FilterByAtomMapNums = False):

 385     matched atom indices present in a pattern molecule. The list of atom indices
 386     correspond to a list retrieved by RDKit function GetSubstructureMatches using
 387     SMILES/SMARTS pattern. The atom indices are optionally filtered by mapping
 388     atom numbers to appropriate atom indices during the generation of the molecule. For
 389     example: [O:1]=[S:2](=[O])[C:3][C:4].
 390      
 391     Arguments:
 392         Mol (object): RDKit molecule object.
 393         PatternMol (object): RDKit molecule object for a SMILES/SMARTS pattern.
 394         AtomIndicesList (list): A list of lists containing atom indices.
 395         FilterByAtomMapNums (bool): Filter matches by atom map numbers.
 396 
 397     Returns:
 398         list : A list of lists containg RDKit molecule objects or None.
 399 
 400     """
 401 
 402     AtomMapIndices = _GetAtomMapIndices(PatternMol) if FilterByAtomMapNums else None

 406         Mols.append(_MolFromSubstructureMatch(Mol, PatternMol, AtomIndices, AtomMapIndices))

 454 def _MolFromSubstructureMatch(Mol, PatternMol, AtomIndices, AtomMapIndices):
 455     """Generate a RDKit molecule object for a list of matched atom indices and available
 456    atom map indices.
 457     """
 458 
 459     if AtomMapIndices is not None:
 460         MatchedAtomIndices = [AtomIndices[Index] for Index in AtomMapIndices]
 461     else:
 462         MatchedAtomIndices = list(AtomIndices)
 463 
 464     return _GetMolFromAtomIndices(Mol, MatchedAtomIndices)

 466 def _GetAtomMapIndices(PatternMol):
 467     """Get a list of any available atom indices corresponding to atom map numbers
 468     present in  pattern SMILES/SMARTS used for creating pattern molecule.
 469     """
 470 
 471     # Setup a atom map number to atom indices map..
 472     AtomMapNumToIndices = {}
 473     for Atom in PatternMol.GetAtoms():
 474         AtomMapNum = Atom.GetAtomMapNum()
 475         
 476         if AtomMapNum:
 477             AtomMapNumToIndices[AtomMapNum] = Atom.GetIdx()
 478     
 479     # Setup atom indices corresponding to atom map numbers...
 480     AtomMapIndices = None
 481     if len(AtomMapNumToIndices):
 482         AtomMapIndices = [AtomMapNumToIndices[AtomMapNum] for AtomMapNum in sorted(AtomMapNumToIndices)]
 483 
 484     return AtomMapIndices

 496 def _GetMolFromAtomIndices(Mol, AtomIndices):
 497     """Generate a RDKit molecule object from atom indices returned by
 498    substructure search.
 499     """
 500 
 501     BondIndices = []
 502     for AtomIndex in AtomIndices:
 503         Atom = Mol.GetAtomWithIdx(AtomIndex)
 504         
 505         for AtomNbr in Atom.GetNeighbors():
 506             AtomNbrIndex = AtomNbr.GetIdx()
 507             if AtomNbrIndex not in AtomIndices:
 508                 continue
 509             
 510             BondIndex = Mol.GetBondBetweenAtoms(AtomIndex, AtomNbrIndex).GetIdx()
 511             if BondIndex in BondIndices:
 512                 continue
 513                 
 514             BondIndices.append(BondIndex)
 515             
 516     MatchedMol = Chem.PathToSubmol(Mol, BondIndices) if len(BondIndices) else None
 517     
 518     return MatchedMol

 521 def ReadAndValidateMolecules(FileName, **KeyWordArgs):
 522     """Read molecules from an input file, validate all molecule objects, and return
 523     a list of valid and non-valid molecule objects along with their counts.
 524     
 525     Arguments:
 526         FileName (str): Name of a file with complete path.
 527         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 528             processing molecules.
 529 
 530     Returns:
 531         list : List of valid RDKit molecule objects.
 532         int : Number of total molecules in input file. 
 533         int : Number of valid molecules in input file. 
 534 
 535     Notes:
 536         The file extension is used to determine type of the file and set up an appropriate
 537         file reader.
 538 
 539     """
 540 
 541     AllowEmptyMols = True
 542     if "AllowEmptyMols" in KeyWordArgs:
 543         AllowEmptyMols = KeyWordArgs["AllowEmptyMols"]
 544     
 545     Mols = ReadMolecules(FileName, **KeyWordArgs)
 546 
 547     if AllowEmptyMols:
 548         ValidMols = [Mol for Mol in Mols if Mol is not None]
 549     else:
 550         ValidMols = []
 551         MolCount = 0
 552         for Mol in Mols:
 553             MolCount += 1
 554             if Mol is None:
 555                 continue
 556             
 557             if IsMolEmpty(Mol):
 558                 MolName = GetMolName(Mol, MolCount)
 559                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 560                 continue
 561             
 562             ValidMols.append(Mol)
 563             
 564     MolCount = len(Mols)
 565     ValidMolCount = len(ValidMols)
 566 
 567     return (ValidMols, MolCount, ValidMolCount)

 569 def ReadMolecules(FileName, **KeyWordArgs):
 570     """Read molecules from an input file without performing any validation
 571     and creation of molecule objects.
 572     
 573     Arguments:
 574         FileName (str): Name of a file with complete path.
 575         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 576             processing molecules.
 577 
 578     Returns:
 579         list : List of RDKit molecule objects.
 580 
 581     Notes:
 582         The file extension is used to determine type of the file and set up an appropriate
 583         file reader.
 584 
 585     """
 586 
 587     # Set default values for possible arguments...
 588     ReaderArgs = {"Sanitize": True, "RemoveHydrogens": True, "StrictParsing": True,  "SMILESDelimiter" : ' ', "SMILESColumn": 1, "SMILESNameColumn": 2, "SMILESTitleLine": True }
 589 
 590     # Set specified values for possible arguments...
 591     for Arg in ReaderArgs:
 592         if Arg in KeyWordArgs:
 593             ReaderArgs[Arg] = KeyWordArgs[Arg]
 594 
 595     # Modify specific valeus for SMILES...
 596     if MiscUtil.CheckFileExt(FileName, "smi csv tsv txt"):
 597         Args = ["Sanitize", "SMILESTitleLine"]
 598         for Arg in Args:
 599             if ReaderArgs[Arg] is True:
 600                 ReaderArgs[Arg] = 1
 601             else:
 602                 ReaderArgs[Arg] = 0
 603     
 604     Mols = []
 605     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 606         return ReadMoleculesFromSDFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 607     elif MiscUtil.CheckFileExt(FileName, "mol"):
 608         return ReadMoleculesFromMolFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 609     elif MiscUtil.CheckFileExt(FileName, "mol2"):
 610         return ReadMoleculesFromMol2File(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 611     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 612         return ReadMoleculesFromPDBFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 613     elif MiscUtil.CheckFileExt(FileName, "smi txt csv tsv"):
 614         SMILESColumnIndex = ReaderArgs["SMILESColumn"] - 1
 615         SMILESNameColumnIndex = ReaderArgs["SMILESNameColumn"] - 1
 616         return ReadMoleculesFromSMILESFile(FileName, ReaderArgs["SMILESDelimiter"], SMILESColumnIndex, SMILESNameColumnIndex, ReaderArgs["SMILESTitleLine"], ReaderArgs["Sanitize"])

 706 def MoleculesWriter(FileName, **KeyWordArgs):
 707     """Set up a molecule writer.
 708     
 709     Arguments:
 710         FileName (str): Name of a file with complete path.
 711         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 712             processing molecules.
 713 
 714     Returns:
 715         RDKit object : Molecule writer.
 716 
 717     Notes:
 718         The file extension is used to determine type of the file and set up an appropriate
 719         file writer.
 720 
 721     """
 722     
 723     # Set default values for possible arguments...
 724     WriterArgs = {"Compute2DCoords" : False, "Kekulize": False, "SMILESDelimiter" : ' ', "SMILESIsomeric": True, "SMILESTitleLine": True, "SMILESMolName": True}
 725 
 726     # Set specified values for possible arguments...
 727     for Arg in WriterArgs:
 728         if Arg in KeyWordArgs:
 729             WriterArgs[Arg] = KeyWordArgs[Arg]
 730     
 731     Writer = None
 732     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 733         Writer = Chem.SDWriter(FileName)
 734         if WriterArgs["Kekulize"]:
 735             Writer.SetKekulize(True)
 736     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 737         Writer = Chem.PDBWriter(FileName)
 738     elif MiscUtil.CheckFileExt(FileName, "smi"):
 739         # Text for the name column in the title line. Blank indicates not to include name column
 740         # in the output file...
 741         NameHeader = 'Name' if WriterArgs["SMILESMolName"] else ''
 742         Writer = Chem.SmilesWriter(FileName, delimiter = WriterArgs["SMILESDelimiter"], nameHeader = NameHeader, includeHeader = WriterArgs["SMILESTitleLine"],  isomericSmiles = WriterArgs["SMILESIsomeric"], kekuleSmiles = WriterArgs["Kekulize"])
 743     else:
 744         MiscUtil.PrintWarning("RDKitUtil.WriteMolecules: Non supported file type: %s" % FileName)
 745     
 746     return Writer
 747     
 748 def WriteMolecules(FileName, Mols, **KeyWordArgs):
 749     """Write molecules to an output file.
 750     
 751     Arguments:
 752         FileName (str): Name of a file with complete path.
 753         Mols (list): List of RDKit molecule objects. 
 754         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 755             processing molecules.
 756 
 757     Returns:
 758         int : Number of total molecules.
 759         int : Number of processed molecules written to output file.
 760 
 761     Notes:
 762         The file extension is used to determine type of the file and set up an appropriate
 763         file writer.
 764 
 765     """
 766     
 767     Compute2DCoords = False
 768     if "Compute2DCoords" in KeyWordArgs:
 769         Compute2DCoords = KeyWordArgs["Compute2DCoords"]
 770     
 771     SetSMILESMolProps = KeyWordArgs["SetSMILESMolProps"] if "SetSMILESMolProps" in KeyWordArgs else False
 772         
 773     MolCount = len(Mols)
 774     ProcessedMolCount = 0
 775     
 776     Writer = MoleculesWriter(FileName, **KeyWordArgs)
 777     
 778     if Writer is None:
 779         return (MolCount, ProcessedMolCount)
 780     
 781     FirstMol = True
 782     for Mol in Mols:
 783         if Mol is None:
 784             continue
 785 
 786         if FirstMol:
 787             FirstMol = False
 788             if SetSMILESMolProps:
 789                 SetWriterMolProps(Writer, Mol)
 790                 
 791         ProcessedMolCount += 1
 792         if Compute2DCoords:
 793             AllChem.Compute2DCoords(Mol)
 794         
 795         Writer.write(Mol)
 796     
 797     Writer.close()
 798     
 799     return (MolCount, ProcessedMolCount)

 801 def SetWriterMolProps(Writer, Mol):
 802     """Setup molecule properties for a writer to output.
 803     
 804     Arguments:
 805         Writer (object): RDKit writer object.
 806         Mol (object): RDKit molecule object.
 807 
 808     Returns:
 809         object : Writer object.
 810 
 811     """

 813     if len(PropNames):
 814         Writer.SetProps(PropNames)
 815         
 816     return Writer
