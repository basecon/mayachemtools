   83 def PerformPositionalAnalogueScan():
   84     """Perform positional analogue scan."""
   85     
   86     # Setup a molecule reader for input file...
   87     MiscUtil.PrintInfo("\nProcessing file %s..." % OptionsInfo["Infile"])
   88     OptionsInfo["InfileParams"]["AllowEmptyMols"] = True
   89     Mols  = RDKitUtil.ReadMolecules(OptionsInfo["Infile"], **OptionsInfo["InfileParams"])
   90 
   91     # Set up a molecule writer...
   92     Writer = RDKitUtil.MoleculesWriter(OptionsInfo["Outfile"], **OptionsInfo["OutfileParams"])
   93     if Writer is None:
   94         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % OptionsInfo["Outfile"])
   95     MiscUtil.PrintInfo("Generating file %s..." % OptionsInfo["Outfile"])
   96 
   97     MolCount, ValidMolCount, SearchPatternMissingCount, AnaloguesGenerationFailedCount, AnaloguesCount = ProcessMolecules(Mols, Writer)
   98 
   99     if Writer is not None:
  100         Writer.close()
  101     
  102     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
  103     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
  104     if not OptionsInfo["RxnSMARTSMode"]:
  105         MiscUtil.PrintInfo("Number of molecules with missing search pattern: %d" % SearchPatternMissingCount)
  106     MiscUtil.PrintInfo("Number of molecules failed during generation of analogues: %d" % AnaloguesGenerationFailedCount)
  107     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + SearchPatternMissingCount + AnaloguesGenerationFailedCount))
  108     
  109     MiscUtil.PrintInfo("\nTotal number of analogues for valid molecules: %d" % AnaloguesCount)
  110     MiscUtil.PrintInfo("Average number of analogues per valid molecule: %d" % (int(AnaloguesCount/(ValidMolCount))))

  112 def ProcessMolecules(Mols, Writer):
  113     """Process molecules to generate analogues."""
  114     
  115     if OptionsInfo["MPMode"]:
  116         return ProcessMoleculesUsingMultipleProcesses( Mols, Writer)
  117     else:
  118         return ProcessMoleculesUsingSingleProcess(Mols, Writer)

  120 def ProcessMoleculesUsingSingleProcess(Mols, Writer):
  121     """Process molecules to generate analogues using a single process."""
  122 
  123     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
  124     SetSMILESMolProps = OptionsInfo["OutfileParams"]["SetSMILESMolProps"]
  125 
  126     (MolCount, ValidMolCount, SearchPatternMissingCount, AnaloguesGenerationFailedCount, AnaloguesCount) = [0] * 5
  127 
  128     # Initialize search and target patterns info...
  129     SetupSearchAndTargetPatternsInfo()
  130 
  131     FirstMol = True
  132     for Mol in Mols:
  133         MolCount += 1
  134         
  135         if Mol is None:
  136             continue
  137         
  138         if RDKitUtil.IsMolEmpty(Mol):
  139             if not OptionsInfo["QuietMode"]:
  140                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
  141                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
  142             continue
  143         ValidMolCount += 1
  144 
  145         MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus = GenerateMolAnalogues(Mol, MolCount)
  146         
  147         if FirstMol:
  148             FirstMol = False
  149             if SetSMILESMolProps:
  150                 if Writer is not None:
  151                     RDKitUtil.SetWriterMolProps(Writer, Mol)
  152         
  153         if not OptionsInfo["RxnSMARTSMode"]:
  154             if SearchPatternMissingStatus:
  155                 SearchPatternMissingCount += 1
  156                 WriteMolecule(Writer, Mol, Compute2DCoords)
  157                 continue
  158         
  159         if not AnaloguesGenerationStatus:
  160             AnaloguesGenerationFailedCount += 1
  161             WriteMolecule(Writer, Mol, Compute2DCoords)
  162             continue
  163         
  164         # Write out molecues and analogues...
  165         WriteMolAnalogues(Writer, Mol, MolCount, MolAnalogues, Compute2DCoords)
  166         
  167         AnaloguesCount += len(MolAnalogues)
  168 
  169     return (MolCount, ValidMolCount, SearchPatternMissingCount, AnaloguesGenerationFailedCount, AnaloguesCount)

  171 def ProcessMoleculesUsingMultipleProcesses(Mols, Writer):
  172     """Process molecules to generate analogues using multiprocessing."""
  173 
  174     MPParams = OptionsInfo["MPParams"]
  175     
  176     # Setup data for initializing a worker process...
  177     MiscUtil.PrintInfo("Encoding options info...")
  178 
  179     InitializeWorkerProcessArgs = (MiscUtil.ObjectToBase64EncodedString(Options), MiscUtil.ObjectToBase64EncodedString(OptionsInfo))
  180     
  181     # Setup a encoded mols data iterable for a worker process...
  182     WorkerProcessDataIterable = RDKitUtil.GenerateBase64EncodedMolStrings(Mols)
  183 
  184     # Setup process pool along with data initialization for each process...
  185     MiscUtil.PrintInfo("\nConfiguring multiprocessing using %s method..." % ("mp.Pool.imap()" if re.match("^Lazy$", MPParams["InputDataMode"], re.I) else "mp.Pool.map()"))
  186     MiscUtil.PrintInfo("NumProcesses: %s; InputDataMode: %s; ChunkSize: %s\n" % (MPParams["NumProcesses"], MPParams["InputDataMode"], ("automatic" if MPParams["ChunkSize"] is None else MPParams["ChunkSize"])))
  187     
  188     ProcessPool = mp.Pool(MPParams["NumProcesses"], InitializeWorkerProcess, InitializeWorkerProcessArgs)
  189     
  190     # Start processing...
  191     if re.match("^Lazy$", MPParams["InputDataMode"], re.I):
  192         Results = ProcessPool.imap(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
  193     elif re.match("^InMemory$", MPParams["InputDataMode"], re.I):
  194         Results = ProcessPool.map(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
  195     else:
  196         MiscUtil.PrintError("The value, %s, specified for \"--inputDataMode\" is not supported." % (MPParams["InputDataMode"]))
  197     
  198     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
  199     SetSMILESMolProps = OptionsInfo["OutfileParams"]["SetSMILESMolProps"]
  200 
  201     (MolCount, ValidMolCount, SearchPatternMissingCount, AnaloguesGenerationFailedCount, AnaloguesCount) = [0] * 5
  202     FirstMol = True
  203     for Result in Results:
  204         MolCount += 1
  205         MolIndex, EncodedMol, EncodedMolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus  = Result
  206         
  207         if EncodedMol is None:
  208             continue
  209         ValidMolCount += 1
  210 
  211         Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
  212         
  213         if FirstMol:
  214             FirstMol = False
  215             if SetSMILESMolProps:
  216                 if Writer is not None:
  217                     RDKitUtil.SetWriterMolProps(Writer, Mol)
  218         
  219         if not OptionsInfo["RxnSMARTSMode"]:
  220             if SearchPatternMissingStatus:
  221                 SearchPatternMissingCount += 1
  222                 WriteMolecule(Writer, Mol, Compute2DCoords)
  223                 continue
  224         
  225         if not AnaloguesGenerationStatus:
  226             AnaloguesGenerationFailedCount += 1
  227             WriteMolecule(Writer, Mol, Compute2DCoords)
  228             continue
  229         
  230         MolAnalogues = [RDKitUtil.MolFromBase64EncodedMolString(EncodedMolAnalogue) for EncodedMolAnalogue in EncodedMolAnalogues]
  231         
  232         # Write out molecues and analogues...
  233         WriteMolAnalogues(Writer, Mol, MolCount, MolAnalogues, Compute2DCoords)
  234         
  235         AnaloguesCount += len(MolAnalogues)
  236     
  237     return (MolCount, ValidMolCount, SearchPatternMissingCount, AnaloguesGenerationFailedCount, AnaloguesCount)

  239 def InitializeWorkerProcess(*EncodedArgs):
  240     """Initialize data for a worker process."""
  241     
  242     global Options, OptionsInfo
  243 
  244     MiscUtil.PrintInfo("Starting process (PID: %s)..." % os.getpid())
  245 
  246     # Decode Options and OptionInfo...
  247     Options = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[0])
  248     OptionsInfo = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[1])
  249     
  250     # Initialize search and target  patterns info...
  251     SetupSearchAndTargetPatternsInfo()

  253 def WorkerProcess(EncodedMolInfo):
  254     """Process data for a worker process."""
  255 
  256     MolIndex, EncodedMol = EncodedMolInfo
  257 
  258     MolAnalogues = None
  259     SearchPatternMissingStatus = False
  260     AnaloguesGenerationStatus = True
  261     
  262     if EncodedMol is None:
  263         return [MolIndex, None, MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus]
  264 
  265     Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
  266     if RDKitUtil.IsMolEmpty(Mol):
  267         if not OptionsInfo["QuietMode"]:
  268             MolName = RDKitUtil.GetMolName(Mol, (MolIndex + 1))
  269             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
  270         return [MolIndex, None, MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus]
  271 
  272     MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus = GenerateMolAnalogues(Mol, (MolIndex +1))
  273 
  274     EncodedMolAnalogues = None
  275     if MolAnalogues is not None:
  276         EncodedMolAnalogues = [RDKitUtil.MolToBase64EncodedMolString(MolAnalogue, PropertyPickleFlags = Chem.PropertyPickleOptions.MolProps | Chem.PropertyPickleOptions.PrivateProps) for MolAnalogue in MolAnalogues]
  277     
  278     return [MolIndex, EncodedMol, EncodedMolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus]

  280 def GenerateMolAnalogues(Mol, MolNum = None):
  281     """Generate analogues."""
  282 
  283     if OptionsInfo["AttachAtomsMode"] or OptionsInfo["ReplaceAtomsMode"] or OptionsInfo["AttachSMILESMode"]:
  284         return GenerateMolAnaloguesForMultipleScanModes(Mol, MolNum)
  285     elif OptionsInfo["RxnSMARTSMode"]:
  286         return GenerateMolAnaloguesForRxnSMARTSMode(Mol, MolNum)
  287     elif OptionsInfo["TandemScanMode"]:
  288         return GenerateMolAnaloguesForTandemScanMode(Mol, MolNum)
  289     else:
  290         MiscUtil.PrintError("Failed to generate analogues.  The mode value, %s, specified using \"-m, --mode\" is not supported." % (OptionsInfo["Mode"]))

  292 def GenerateMolAnaloguesForMultipleScanModes(Mol, MolNum = None):
  293     """Generate analogues for the following scan modes: AttachAtom, ReplaceAtom
  294     or AttachSMILES."""
  295 
  296     MolAnalogues = None
  297     SearchPatternMissingStatus = False
  298     AnaloguesGenerationStatus = True
  299 
  300     # Retrieve matched atoms...
  301     MolMatchedAtomIndices = SetupMolMatchedAtomIndices(Mol, MolNum)
  302     if MolMatchedAtomIndices is None:
  303         SearchPatternMissingStatus = True
  304         return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)
  305 
  306     AtomicNumber = OptionsInfo["TargetPatternAtomicNumber"]
  307     SMILESMol = OptionsInfo["TargetPatternSMILESMol"]
  308     
  309     MolAnalogues = []
  310     MolAnaloguesSet = set()
  311     for AtomIndicesCombination in combinations(MolMatchedAtomIndices, OptionsInfo["ComboSearchAtoms"]):
  312         MolAnalogue = Chem.RWMol(Mol)
  313 
  314         for AtomIndex in AtomIndicesCombination:
  315             if OptionsInfo["AttachAtomsMode"]:
  316                 MolAnalogue = AttachAtom(MolAnalogue, AtomIndex, AtomicNumber, BondOrder = Chem.rdchem.BondType.SINGLE)
  317             elif OptionsInfo["ReplaceAtomsMode"]:
  318                 MolAnalogue = ReplaceAtom(MolAnalogue, AtomIndex, AtomicNumber)
  319             elif OptionsInfo["AttachSMILESMode"]:
  320                 AttachAtomIndex = MolAnalogue.GetNumAtoms()
  321                 MolAnalogue = AttachMol(MolAnalogue, AtomIndex, SMILESMol, AttachAtomIndex, BondOrder = Chem.rdchem.BondType.SINGLE)
  322             
  323         if not SanitizeMolAnalogue(MolAnalogue, MolNum):
  324             continue
  325 
  326         TrackAnalogues(MolAnalogue, MolAnaloguesSet, MolAnalogues)
  327         
  328     if len(MolAnalogues) == 0:
  329         AnaloguesGenerationStatus = False
  330         
  331     return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)
  332 
  333 def GenerateMolAnaloguesForRxnSMARTSMode(Mol, MolNum = None):
  334     """Generate analogues for reaction SMARTS mode."""
  335 
  336     MolAnalogues = None
  337     SearchPatternMissingStatus = False
  338     AnaloguesGenerationStatus = True
  339     
  340     Rxn = OptionsInfo["TargetPatternRxn"]
  341     RxnProducts = Rxn.RunReactants((Mol,))
  342 
  343     if not len(RxnProducts):
  344         MolAnalogues = None
  345         AnaloguesGenerationStatus = False
  346         return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)
  347 
  348     MolAnalogues = []
  349     MolAnaloguesSet = set()
  350     for RxnProduct in RxnProducts:
  351         MolAnalogue = RxnProduct[0]
  352         
  353         if not SanitizeMolAnalogue(MolAnalogue, MolNum):
  354             continue
  355         
  356         TrackAnalogues(MolAnalogue, MolAnaloguesSet, MolAnalogues)
  357     
  358     if len(MolAnalogues) == 0:
  359         AnaloguesGenerationStatus = False
  360         
  361     return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)

  363 def GenerateMolAnaloguesForTandemScanMode(Mol, MolNum = None):
  364     """Generate analogues for tandem scan mode."""
  365 
  366     MolAnalogues = None
  367     SearchPatternMissingStatus = False
  368     AnaloguesGenerationStatus = True
  369 
  370     # Retrieve matched atoms...
  371     MolMatchedAtomIndices = SetupMolMatchedAtomIndices(Mol, MolNum)
  372     if MolMatchedAtomIndices is None:
  373         SearchPatternMissingStatus = True
  374         return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)
  375 
  376     TargetPatternTandemScanInfo = OptionsInfo["TargetPatternTandemScanInfo"]
  377     NumOfTandemOperations = len(TargetPatternTandemScanInfo["Names"])
  378     if  len(MolMatchedAtomIndices) < NumOfTandemOperations:
  379         if not OptionsInfo["QuietMode"]:
  380             MolName = RDKitUtil.GetMolName(Mol, MolNum)
  381             MiscUtil.PrintWarning("Number of matched search pattern atoms, %s, must be >= number of specified operations, %s, during TandemScan mode for molecule %s..." % (len(MolMatchedAtomIndices), NumOfTandemOperations, MolName))
  382         SearchPatternMissingStatus = True
  383         return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)
  384     
  385     MolAnalogues = []
  386     MolAnaloguesSet = set()
  387     for AtomIndicesCombination in combinations(MolMatchedAtomIndices, NumOfTandemOperations):
  388         AtomIndicesPermutations = []
  389         if OptionsInfo["PermuteTandemScan"]:
  390             for AtomIndicesPermutation in permutations(AtomIndicesCombination, NumOfTandemOperations):
  391                 AtomIndicesPermutations.append(list(AtomIndicesPermutation))
  392         else:
  393             AtomIndicesPermutations.append(list(AtomIndicesCombination))
  394         
  395         for AtomIndicesPermutation in AtomIndicesPermutations:
  396             MolAnalogue = Chem.RWMol(Mol)
  397             
  398             for NameIndex, Name in enumerate(TargetPatternTandemScanInfo["Names"]):
  399                 AtomIndex = AtomIndicesPermutation[NameIndex]
  400                 
  401                 if re.match("^AttachAtoms$", Name, re.I):
  402                     MolAnalogue = AttachAtom(MolAnalogue, AtomIndex, TargetPatternTandemScanInfo["PatternAtomicNumber"][Name], BondOrder = Chem.rdchem.BondType.SINGLE)
  403                 elif re.match("^ReplaceAtoms$", Name, re.I):
  404                     MolAnalogue = ReplaceAtom(MolAnalogue, AtomIndex, TargetPatternTandemScanInfo["PatternAtomicNumber"][Name])
  405                 elif re.match("^AttachSMILES$", Name, re.I):
  406                     AttachAtomIndex = MolAnalogue.GetNumAtoms()
  407                     MolAnalogue = AttachMol(MolAnalogue, AtomIndex, TargetPatternTandemScanInfo["PatternSMILESMol"][Name], AttachAtomIndex, BondOrder = Chem.rdchem.BondType.SINGLE)
  408                     MolAnalogue = Chem.RWMol(MolAnalogue)
  409             
  410             if not SanitizeMolAnalogue(MolAnalogue, MolNum):
  411                 continue
  412             
  413             TrackAnalogues(MolAnalogue, MolAnaloguesSet, MolAnalogues)
  414     
  415     if len(MolAnalogues) == 0:
  416         AnaloguesGenerationStatus = False
  417         
  418     return (MolAnalogues, SearchPatternMissingStatus, AnaloguesGenerationStatus)
  419 
  420 def TrackAnalogues(MolAnalogue, MolAnaloguesSet, MolAnalogues):
  421     """Track analogues. """
  422     
  423     if OptionsInfo["UniqueAnalogues"]:
  424         AnalogueSMILES = Chem.MolToSmiles(Chem.RemoveHs(MolAnalogue))
  425         if AnalogueSMILES not in MolAnaloguesSet:
  426             MolAnaloguesSet.add(AnalogueSMILES)
  427             MolAnalogues.append(MolAnalogue)
  428     else:
  429         MolAnalogues.append(MolAnalogue)
  430     
  431 def ReplaceAtom(Mol, AtomIndex, AtomicNumber):
  432     """Replace an existing atom with a new atom in a molecule."""
  433     
  434     Atom = Mol.GetAtomWithIdx(AtomIndex)
  435     Atom.SetAtomicNum(AtomicNumber)
  436 
  437     return Mol
  438 
  439 def AttachAtom(Mol, AtomIndex, AtomicNumber, BondOrder = Chem.rdchem.BondType.SINGLE):
  440     """Attach atom to an existing atom in a molecule."""
  441     
  442     NewAtomIndex = Mol.AddAtom(Chem.Atom(AtomicNumber))
  443     Mol.AddBond(AtomIndex, NewAtomIndex, BondOrder)
  444 
  445     return Mol
  446 
  447 def AttachMol(Mol, AtomIndex, AttachMol, AttachAtomIndex, BondOrder = Chem.rdchem.BondType.SINGLE):
  448     """Attach molecule to an exisiting molecule."""
  449     
  450     NewMol = Chem.CombineMols(Mol, AttachMol)
  451     
  452     NewMol = Chem.EditableMol(NewMol)
  453     NewMol.AddBond(AtomIndex, AttachAtomIndex, order = BondOrder)
  454     NewMol = NewMol.GetMol()
  455     
  456     return NewMol

  458 def SanitizeMolAnalogue(Mol, MolNum):
  459     """Sanitize analogue and catch any errors. """
  460     
  461     Status = True
  462     try:
  463         Chem.SanitizeMol(Mol)
  464     except Exception as ErrMsg:
  465         if not OptionsInfo["QuietMode"]:
  466             MolName = RDKitUtil.GetMolName(Mol, MolNum)
  467             MiscUtil.PrintInfo("\n%s\nFailed to sanitize analogue of molecule %s...\nIgnoring analogue..." % (ErrMsg, MolName))
  468         Status = False
  469 
  470     return Status

  472 def SetupMolMatchedAtomIndices(Mol, MolNum):
  473     """Setup a list of matched atom indices corresponding to first atom in search
  474     SMARTS pattern for a molecule."""
  475 
  476     MatchedAtoms = []
  477     
  478     # Retrieve first atom match for PAS...
  479     MatchedAtomIndices = [AtomIndices[0] for AtomIndices in Mol.GetSubstructMatches(OptionsInfo["SearchPatternMol"])]
  480     if len(MatchedAtomIndices) == 0:
  481         if not OptionsInfo["QuietMode"]:
  482             MolName = RDKitUtil.GetMolName(Mol, MolNum)
  483             MiscUtil.PrintWarning("No search pattern match for molecule %s..." % MolName)
  484         return None
  485     
  486     return MatchedAtomIndices

  488 def WriteMolAnalogues(Writer, Mol, MolNum, MolAnalogues, Compute2DCoords):
  489     """Write molecule analogues. """
  490 
  491     # Write out molecule...
  492     WriteMolecule(Writer, Mol, Compute2DCoords)
  493     
  494     if MolAnalogues is None:
  495         return
  496 
  497     # Write out analogues...
  498     for Index, MolAnalogue in enumerate(MolAnalogues):
  499         MolName = RDKitUtil.GetMolName(Mol, MolNum)
  500         SetMolAnalogueName(MolAnalogue, MolName, (Index + 1))
  501 
  502         WriteMolecule(Writer, MolAnalogue, Compute2DCoords)
  503     
  504 def SetMolAnalogueName(MolAnalogue, MolName, AnalogueCount):
  505     """Set analogue name."""
  506     
  507     NameSuffix = OptionsInfo["NameSuffix"]
  508     AnalogueName = "%s_%s%d" % (MolName, NameSuffix, AnalogueCount)
  509     MolAnalogue.SetProp("_Name", AnalogueName)
  510 
  511 def WriteMolecule(Writer, Mol, Compute2DCoords):
  512     """Write out molecule."""
  513     
  514     if Compute2DCoords:
  515         AllChem.Compute2DCoords(Mol)
  516     
  517     Writer.write(Mol)

  519 def SetupSearchAndTargetPatternsInfo():
  520     """Setup search and target patterns info."""
  521 
  522     # Setup search pattern molecule...
  523     SearchPattern = OptionsInfo["SearchPattern"]
  524     SearchPatternMol = None
  525     if SearchPattern is not None:
  526         SearchPatternMol = Chem.MolFromSmarts(SearchPattern)
  527         if SearchPatternMol is None:
  528             MiscUtil.PrintError("Failed to create search pattern molecule. The torsion SMILES/SMARTS pattern, \"%s\", specified using \"-s, --searchPattern\" option is not valid." % (SearchPattern))
  529     OptionsInfo["SearchPatternMol"] = SearchPatternMol
  530 
  531     # Setup target pattern information...
  532     Mode = OptionsInfo["Mode"]
  533     TargetPattern = OptionsInfo["TargetPattern"]
  534     
  535     OptionsInfo["TargetPatternAtomicNumber"] = None
  536     OptionsInfo["TargetPatternSMILESMol"] = None
  537     OptionsInfo["TargetPatternRxn"] = None
  538     
  539     if re.match("^(AttachAtoms|ReplaceAtoms)$", Mode, re.I):

  605     
  606     OptionsInfo["UniqueAnalogues"] = True if re.match("^yes$", Options["--uniqueAnalogues"], re.I) else False

  608     # Process and validate search pattern...
  609     SearchPattern = re.sub(" ", "", Options["--searchPattern"])
  610     if not SearchPattern:
  611         MiscUtil.PrintError("No value specified using \"-s, --searchPattern\" option." % (SearchPattern))
  612     if re.match("^auto$", SearchPattern, re.I):
  613         if re.match("^(AttachAtoms|ReplaceAtoms|AttachSMILES|TandemScan)$", Mode, re.I):
  614             SearchPattern = "[cH]"
  615         else:
  616             SearchPattern = None
  617     else:
  618         if re.match("^RxnSMARTS$", Mode, re.I):
  619             MiscUtil.PrintError("The specification of SMARTS search pattern, %s,  using \"-s, --searchPattern\" option is not allowed during, %s, value of \"-m, --mode\"." % (SearchPattern, Mode))

  621     if SearchPattern is not None:
  622         SearchPatternMol = Chem.MolFromSmarts(SearchPattern)
  623         if SearchPatternMol is None:
  624             MiscUtil.PrintError("Failed to create search pattern molecule. The torsion SMILES/SMARTS pattern, \"%s\", specified using \"-s, --searchPattern\" option is not valid." % (SearchPattern))
  625     OptionsInfo["SearchPattern"] = SearchPattern
  626         
  627     # Process and validate target pattern...
  628     TargetPatternTandemScanInfo = None
  629     TargetPattern = re.sub(" ", "", Options["--targetPattern"])
  630     if not TargetPattern:
  631         MiscUtil.PrintError("No value specified using \"-t, --targetPattern\" option." % (TargetPattern))
  632         
  633     if re.match("^auto$", TargetPattern, re.I):
  634         # Setup default target pattern...
  635         if re.match("^AttachAtoms$", Mode, re.I):
  636             TargetPattern = "F"
  637         elif re.match("^ReplaceAtoms$", Mode, re.I):
  638             TargetPattern = "N"
  639         elif re.match("^AttachSMILES$", Mode, re.I):
  640             TargetPattern = "C(F)(F)(F)"
  641         elif re.match("^RxnSMARTS$", Mode, re.I):
  642             TargetPattern = "[cH:1]>>[N:1]"
  643         elif re.match("^TandemScan$", Mode, re.I):
  644             TargetPattern = "ReplaceAtoms,N,AttachAtoms,F"
  645         else:
  646             MiscUtil.PrintError("Failed to setup default target pattern for mode, %s, specified using \"-m, --mode\"" % (Mode))
  647     
  648     if re.match("^(AttachAtoms|ReplaceAtoms)$", Mode, re.I):
  649         if not RDKitUtil.IsValidElementSymbol(TargetPattern):
  650             MiscUtil.PrintError("The target pattern, \"%s\", specified using \"-t, --targetPattern\" must be a valid atomic symbol during, %s, value of \"-m, --mode\" option." % (TargetPattern, Mode))
  651     elif re.match("^AttachSMILES$", Mode, re.I):
  652         ValidateSMILESTargetPattern(TargetPattern, Mode)
  653     elif re.match("^RxnSMARTS$", Mode, re.I):
  654         try:
  655             Rxn = AllChem.ReactionFromSmarts(TargetPattern)
  656         except Exception as ErrMsg:
  657             MiscUtil.PrintError("The target pattern, \"%s\", specified using \"-t, --targetPattern\" must be a valid RxnSMARTS  during, %s, value of \"-m, --mode\" option.\nErrMsg: %s" % (TargetPattern, Mode, ErrMsg))
  658     elif re.match("^TandemScan$", Mode, re.I):
  659         TargetPatternTandemScanInfo = {}
  660         TargetPatternTandemScanInfo["Names"] = []
  661         TargetPatternTandemScanInfo["Pattern"] = {}
  662         
  663         TargetPatternWords = TargetPattern.split(",")
  664         if len(TargetPatternWords) % 2:
  665             MiscUtil.PrintError("The number of comma delimited paramater names and values, %d, specified using \"-t --targetPattern\" option must be an even number during, %s, value of \"-m, --mode\" option." % (len(TargetPatternWords), Mode))
  666         
  667         ValidOperationNames = ["AttachAtoms", "ReplaceAtoms", "AttachSMILES"]
  668         CanonicalOperationNamesMap = {}
  669         for OperationName in ValidOperationNames:
  670             CanonicalOperationNamesMap[OperationName.lower()] = OperationName
  671         
  672         # Validate PAS operation name and value pairs...
  673         for Index in range(0, len(TargetPatternWords), 2):
  674             OperationName = TargetPatternWords[Index]
  675             OperationPattern = TargetPatternWords[Index + 1]
  676         
  677             CanonicalOperationName = OperationName.lower()
  678             if  not CanonicalOperationName in CanonicalOperationNamesMap:
  679                 MiscUtil.PrintError("The operation name, %s, specified using \"-t, --targetPattern\" is not a valid during, %s, value of \"-m, --mode\" option. Supported operation  names: %s" % (OperationName, Mode, " ".join(ValidOperationNames)))
  680             
  681             if not OperationPattern:
  682                 MiscUtil.PrintError("Empty target pattern specified using \"-t, --targetPattern\" must be a valid atomic symbol for operation, %s, during, %s, value of \"-m, --mode\" option." % (OperationName, Mode))
  683             
  684             if re.match("^(AttachAtoms|ReplaceAtoms)$", OperationName, re.I):
  685                 if not RDKitUtil.IsValidElementSymbol(OperationPattern):
  686                     MiscUtil.PrintError("The target pattern, \"%s\", specified using \"-t, --targetPattern\" must be a valid atomic symbol for operation, %s, during, %s, value of \"-m, --mode\" option." % (OperationPattern, OperationName, Mode))
  687             elif re.match("^AttachSMILES$", OperationName, re.I):
  688                 ValidateSMILESTargetPattern(OperationPattern, Mode, OperationName)
  689             
  690             # Track info...
  691             Name = CanonicalOperationNamesMap[CanonicalOperationName]
  692             TargetPatternTandemScanInfo["Names"].append(Name)
  693             TargetPatternTandemScanInfo["Pattern"][Name] = OperationPattern
  694     
  695     OptionsInfo["TargetPattern"] = TargetPattern
  696     OptionsInfo["TargetPatternTandemScanInfo"] = TargetPatternTandemScanInfo
